Enum Definitions

Introduction#

In June of 1997 the first edition of ECMA-262, "ECMAScript: A general purpose, cross-platform programming language", was published as a standard. Within that document, section "7.4.3 Future Reserved Words" described "The following words are used as keywords in proposed extensions and are therefore reserved to allow for the possibility of future adoption of those extensions." and included the enum token.

Many other languages provide some form of "Enumerated Type" (or, enumerator).

The enumerator is a data type containing a set of elements. An element is an identifier for a constant value. Each of the enumerator's value will be unique from each other, in this case they are all Symbols. The value does not need to be explicitly assigned, the String Value of the Identifier is used by default.

1.1Static Semantics: Early Errors#

EnumDeclaration[Yield]:enumBindingIdentifier[?Yield]{EnumElementList}

  • If the source code matching this production is strict code, it is a Syntax Error if BindingIdentifier is the IdentifierName eval or the IdentifierName arguments.

EnumElement[Yield]:IdentifierName[?Yield]

  • It is a Syntax Error if the StringValue of IdentifierName is "eval" or "arguments".

EnumLexicalDeclaration[Yield]:enum{EnumElementList}

  • It is a Syntax Error if the EnumKeyName of any element of the EnumElementList also occurs in the LexicallyDeclaredNames of ____
or LexicallyScopedDeclarations?

1.2Static Semantics: BoundNames#

EnumDeclaration[Yield]:enumBindingIdentifier[?Yield]{EnumElementList}

  1. Return the BoundNames of BindingIdentifier.

EnumElementList[Yield]:EnumElement[?Yield]
EnumElementList[Yield]:EnumElementList[?Yield]EnumElement[?Yield]

  1. Return the a new List containing the entries in EnumKeyNameList of of EnumElementList.

1.3Static Semantics: EnumKeyName#

EnumElement[Yield]:IdentifierName[?Yield]

  1. Return StringValue of IdentifierName.

1.4Static Semantics: EnumKeyNameList#

EnumElementList[Yield]:EnumElement[?Yield]

  1. Return a new List containing EnumKeyName of EnumElement.

EnumElementList[Yield]:EnumElementList[?Yield]EnumElement[?Yield]

  1. Let list be EnumKeyNameList of EnumElementList.
  2. Append EnumKeyName of EnumElement to the end of list.
  3. Return list.

function DefineEnumeration(target, isQualified, names, values) {
  let configurable = false;
  let writable = false;
  let enumerable = true;
  let descriptors = Object.create(null);
  let k = 0;

  for (let name of names) {
    let value = Reflect.has(values, name) ? values[name] : k;
    let binding = isQualified ? name : Symbol(name);

    descriptors[binding] = {
      value,
      configurable,
      enumerable,
      writable,
    };

    k++;
  }

  Object.seal(Object.freeze(Object.defineProperties(target, descriptors)));
}


// enum expression
let Days = enum {
  SUNDAY,
  MONDAY,
  TUESDAY,
  WEDNESDAY,
  THURSDAY,
  FRIDAY,
  SATURDAY,
};

// approximately...
let Days;
{
  Days = Object.create(null);
  let isQualified = true;
  let names = [
    "SUNDAY",
    "MONDAY",
    "TUESDAY",
    "WEDNESDAY",
    "THURSDAY",
    "FRIDAY",
    "SATURDAY",
  ];
  let values = {};

  DefineEnumeration(Object.create(null), isQualified, names, values);
}

console.log(Days.SUNDAY); // 0
console.log(Days.SATURDAY); // 6

  SATURDAY,

let Days;
{
  Days = Object.create(null);
  let isQualified = true;
  let names = [
    "SUNDAY",
    "MONDAY",
    "TUESDAY",
    "WEDNESDAY",
    "THURSDAY",
    "FRIDAY",

    TUESDAY: 0

// enum lexical declaration (containing an element with explicitly assigned value)
enum {
  SUNDAY,
  MONDAY,
  TUESDAY,
  WEDNESDAY,
  THURSDAY,
  FRIDAY,
  SATURDAY,
  FUNDAY = "IS NOT A REAL DAY",
}

DefineEnumeration(
  LexicalEnvironment,
  false,
  [
    "SUNDAY",
    "MONDAY",
    "TUESDAY",
    "WEDNESDAY",
    "THURSDAY",
    "FRIDAY",
    "SATURDAY",
    "FUNDAY",
  ],
  {
    FUNDAY: "IS NOT A REAL DAY",
  }
);

console.log(SUNDAY); // 0
console.log(FUNDAY); // "IS NOT A REAL DAY"
